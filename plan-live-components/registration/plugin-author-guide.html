Leaf · Registration

Plugin author guide

Six steps. Define the schema, the op union, the codec, the LLM tools, the renderer, and call the register function. Ship as an npm package with the live-host as a peer dep.

Requirements

  1. An engineer who has never seen this app before can ship a new live component by following this page.
  2. The only contracts the author needs to know are LiveComponentManifest and the React renderer props.

0. File layout

my-extension/
├── package.json                  peerDeps + "sideEffects": ["./dist/index.js"]
├── tsconfig.json
└── src/
    ├── schema.ts                 doc + op types and Zod schemas
    ├── reducer.ts                pure (state, op) → state
    ├── tools.ts                  AI SDK tool factories
    ├── renderer.tsx              React component
    └── index.ts                  manifest + registerLiveComponent call

1. Define the schema

// src/schema.ts
import { z } from "zod";

export const TodoItem = z.object({
  id: z.string().uuid(),
  text: z.string().min(1),
  done: z.boolean().default(false),
});
export type TodoItem = z.infer<typeof TodoItem>;

export const Todo = z.object({ items: z.array(TodoItem) });
export type Todo = z.infer<typeof Todo>;

2. Define the op union + reducer

// src/schema.ts (continued)
export const TodoOp = z.discriminatedUnion("kind", [
  z.object({ kind: z.literal("add-item"),    id: z.string().uuid(), text: z.string() }),
  z.object({ kind: z.literal("toggle-item"), id: z.string().uuid() }),
  z.object({ kind: z.literal("edit-item"),   id: z.string().uuid(), text: z.string() }),
  z.object({ kind: z.literal("remove-item"), id: z.string().uuid() }),
]);
export type TodoOp = z.infer<typeof TodoOp>;
// src/reducer.ts
import type { Todo, TodoOp } from "./schema";

export function reducer(state: Todo, op: TodoOp): Todo {
  switch (op.kind) {
    case "add-item":
      return { ...state, items: [...state.items, { id: op.id, text: op.text, done: false }] };
    case "toggle-item":
      return { ...state,
        items: state.items.map(i => i.id === op.id ? { ...i, done: !i.done } : i) };
    case "edit-item":
      return { ...state,
        items: state.items.map(i => i.id === op.id ? { ...i, text: op.text } : i) };
    case "remove-item":
      return { ...state, items: state.items.filter(i => i.id !== op.id) };
  }
}

3. Write the LLM tools

// src/tools.ts
import { tool } from "ai";
import { z } from "zod";
import type { LiveTool } from "@agent-ui/live-host";
import type { TodoOp } from "./schema";

export function createTodoTools(docId: string): Record<string, LiveTool<TodoOp>> {
  return {
    "todo.addItem": tool({
      description: "Add a new item to the todo list.",
      inputSchema: z.object({ text: z.string().min(1) }),
      execute: async ({ text }) => ({
        ops: [{ kind: "add-item", id: crypto.randomUUID(), text }],
        docId,
      }),
    }),
    "todo.toggleItem": tool({
      description: "Toggle the done state of an item.",
      inputSchema: z.object({ id: z.string().uuid() }),
      execute: async ({ id }) => ({ ops: [{ kind: "toggle-item", id }], docId }),
    }),
    // edit, remove …
  };
}

The tool returns { ops, docId } — never new state. The live-host runtime takes the ops, validates with the manifest's opCodec, applies via the server-side reducer, and broadcasts op-applied.

4. Write the renderer

// src/renderer.tsx
import { useState } from "react";
import type { LiveRendererProps } from "@agent-ui/live-host";
import type { Todo, TodoOp } from "./schema";

export function TodoListView({ doc, dispatch, meta }: LiveRendererProps<Todo, TodoOp>) {
  const [newText, setNewText] = useState("");
  return (
    <div className="rounded-lg border p-3">
      <ul>
        {doc.items.map(item => (
          <li key={item.id} className="flex items-center gap-2 py-1">
            <input
              type="checkbox"
              checked={item.done}
              onChange={() => dispatch({ kind: "toggle-item", id: item.id })}
            />
            <input
              defaultValue={item.text}
              onBlur={(e) => e.target.value !== item.text &&
                dispatch({ kind: "edit-item", id: item.id, text: e.target.value })}
              className={item.done ? "line-through opacity-60" : ""}
            />
            <button onClick={() => dispatch({ kind: "remove-item", id: item.id })}>×</button>
          </li>
        ))}
      </ul>
      <form
        onSubmit={(e) => { e.preventDefault();
          if (!newText.trim()) return;
          dispatch({ kind: "add-item", id: crypto.randomUUID(), text: newText.trim() });
          setNewText("");
        }}
      >
        <input value={newText} onChange={(e) => setNewText(e.target.value)} placeholder="add…" />
      </form>
      {meta.pending.length > 0 && <p>Saving {meta.pending.length} change(s)…</p>}
    </div>
  );
}

The renderer is a normal React component. It receives doc (current state), dispatch (op submit), and meta (docId, pending ops, server seq, status). It never imports the chat core; it never knows whether the LLM or the user originated a change.

5. Compose the manifest and register

// src/index.ts
import { registerLiveComponent, zodCodec } from "@agent-ui/live-host";
import { Todo, TodoOp } from "./schema";
import { reducer } from "./reducer";
import { TodoListView } from "./renderer";

registerLiveComponent<Todo, TodoOp>({
  kind: "todo-list",
  schemaVersions: [1],
  initialState: () => ({ schemaVersion: 1, payload: { items: [] } }),
  reducer,
  opCodec: zodCodec(TodoOp),
  // tools are passed at agent setup time, not here — see tools.ts
  tools: {},
  renderer: TodoListView,
  mockServerReducer: reducer,        // same pure function
});

// Re-export the tool factory so the host can pull it for agent setup.
export { createTodoTools } from "./tools";

That's the entire extension. Consumer host:

// app/src/extensions/index.ts
import "@my-org/live-todo-list";                              // registers renderer
export { createTodoTools } from "@my-org/live-todo-list";     // tool factory

And in the chat-screen agent setup:

// app/src/screens/ChatScreen.tsx
import { createTodoTools } from "../extensions";
const tools = { ...createTodoTools(currentDocId) };
useChat({ /* …, tools */ });

Three lines in the host plus one new npm dep. Removing the extension reverses both. The chat core sees zero changes.
